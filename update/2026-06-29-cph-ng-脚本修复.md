## 问题描述

### 问题 1：语言选择对话框无内容

**现象**：使用 Competitive Companion 导入题目时，弹出的语言选择对话框为空，点击后无反应，最终显示 "Cancelled - 用户脚本未返回数组"

**根本原因**：`ui.chooseItem` API 参数类型错误

```javascript
// ❌ 错误：传入对象数组
ui.chooseItem([
    { label: 'C++', description: 'C++ 语言' },
    { label: 'Python', description: 'Python 语言' }
])

// ✅ 正确：传入字符串数组
ui.chooseItem(['C++', 'Python'], 'Choose language')
```

**官方 API 定义**：
```typescript
interface UI {
  chooseItem(items: string[], placeHolder?: string): Promise<string | undefined>;
}
```

### 问题 2：workspaceFolders 属性错误

**现象**：脚本无法获取正确的项目根目录路径

**根本原因**：`workspaceFolders[0]` 的属性是 `.path`，不是 `.uri.fsPath`

```javascript
// ❌ 错误
const root = workspaceFolders[0].uri.fsPath;

// ✅ 正确
const root = workspaceFolders[0].path;
```

### 问题 3：模板应用问题

**现象**：Python 文件创建时应用的是 C++ 模板

**根本原因**：`cph-ng.problem.templateFile` 配置项只能设置一个模板文件，无法根据语言自动切换

**解决方案**：使用 `template-watcher.js` 作为安全网，在文件创建后自动检测并应用正确的模板

### 修改后的脚本

#### scripts/cph-path-script.js

```javascript
async function process() {
    const lang = await ui.chooseItem(
        ['C++', 'Python'],
        'Choose language'
    );

    if (!lang) return 'Cancelled';

    const isCpp = lang === 'C++';
    const ext = isCpp ? '.cpp' : '.py';

    const root = workspaceFolders[0].path;
    const targetDir = path.join(root, 'src', isCpp ? 'cpp' : 'py');

    const results = [];
    for (const problem of problems) {
        const name = problem.name.replace(/[\\/:*?"<>|]/g, '_');
        const filePath = path.join(targetDir, name + ext);
        results.push(filePath);
    }

    return results;
}
```

#### scripts/template-watcher.js

```javascript
const fs = require('fs');
const path = require('path');

const rootDir = path.dirname(__dirname);
const cppTemplate = fs.readFileSync(path.join(rootDir, 'templates', 'TEMPLATE-2.cpp'), 'utf-8');
const pyTemplate = fs.readFileSync(path.join(rootDir, 'templates', 'TEMPLATE-PY.py'), 'utf-8');

const cppDir = path.join(rootDir, 'src', 'cpp');
const pyDir = path.join(rootDir, 'src', 'py');

function applyTemplate(filePath) {
    const ext = path.extname(filePath);
    
    if (ext === '.cpp') {
        const content = fs.readFileSync(filePath, 'utf-8');
        if (!content.includes('TEMPLATE-2')) {
            fs.writeFileSync(filePath, cppTemplate);
            console.log(`Applied C++ template to: ${filePath}`);
        }
    } else if (ext === '.py') {
        const content = fs.readFileSync(filePath, 'utf-8');
        if (!content.includes('TEMPLATE-PY')) {
            fs.writeFileSync(filePath, pyTemplate);
            console.log(`Applied Python template to: ${filePath}`);
        }
    }
}

function watchDir(dir) {
    if (!fs.existsSync(dir)) {
        fs.mkdirSync(dir, { recursive: true });
    }
    
    fs.watch(dir, (eventType, filename) => {
        if (eventType === 'rename' && filename) {
            const filePath = path.join(dir, filename);
            setTimeout(() => {
                if (fs.existsSync(filePath)) {
                    applyTemplate(filePath);
                }
            }, 100);
        }
    });
    
    console.log(`Watching directory: ${dir}`);
}

watchDir(cppDir);
watchDir(pyDir);

console.log('\nTemplate watcher is running...');
console.log('Press Ctrl+C to stop');

process.on('SIGINT', () => {
    console.log('\nStopping template watcher...');
    process.exit(0);
});
```

## 工作流程

```
1. 打开 VS Code（自动启动 template-watcher.js）
        ↓
2. Competitive Companion 发送题目数据
        ↓
3. cph-ng 调用 customPathScript (cph-path-script.js)
        ↓
4. 脚本弹出对话框选择语言：['C++', 'Python']
        ↓
5. 用户选择后，脚本返回路径数组：
   - C++ → ['src/cpp/题目1.cpp', 'src/cpp/题目2.cpp', ...]
   - Python → ['src/py/题目1.py', 'src/py/题目2.py', ...]
        ↓
6. cph-ng 根据返回的路径创建文件（默认应用 TEMPLATE-2.cpp）
        ↓
7. template-watcher.js 监控到新文件创建
        ↓
8. 自动检测文件扩展名并应用正确模板：
   - .cpp 文件：保持 C++ 模板（已正确）
   - .py 文件：覆盖为 Python 模板
```

## 回滚方案

### 方案 1：恢复旧脚本

将 `scripts/cph-path-script.js` 恢复为修改前的内容（使用对象数组）：

```javascript
async function process() {
    const lang = await ui.chooseItem([
        { label: 'C++', description: 'C++ 语言' },
        { label: 'Python', description: 'Python 语言' }
    ]);
    
    if (!lang) return 'Cancelled';
    
    const isCpp = lang.label === 'C++';
    const ext = isCpp ? '.cpp' : '.py';
    
    const root = workspaceFolders[0].uri.fsPath;
    const targetDir = path.join(root, 'src', isCpp ? 'cpp' : 'py');
    
    const fs = require('fs');
    if (!fs.existsSync(targetDir)) {
        fs.mkdirSync(targetDir, { recursive: true });
    }
    
    const templateName = isCpp ? 'TEMPLATE-2.cpp' : 'TEMPLATE-PY.py';
    const templatePath = path.join(root, 'templates', templateName);
    const templateContent = fs.readFileSync(templatePath, 'utf-8');
    
    const results = [];
    for (const problem of problems) {
        const name = problem.name.replace(/[\\/:*?"<>|]/g, '_');
        const filePath = path.join(targetDir, name + ext);
        fs.writeFileSync(filePath, templateContent);
        results.push(filePath);
    }
    
    return results;
}
```

### 方案 2：简化为固定模板

如果你更倾向于手动切换模板，可以将脚本改为不选择语言：

```javascript
async function process() {
    const root = workspaceFolders[0].path;
    const targetDir = path.join(root, 'src', 'cpp');
    
    const results = [];
    for (const problem of problems) {
        const name = problem.name.replace(/[\\/:*?"<>|]/g, '_');
        const filePath = path.join(targetDir, name + '.cpp');
        results.push(filePath);
    }
    
    return results;
}
```

然后使用 `switch-template.ps1` 手动切换模板：
```powershell
.\scripts\switch-template.ps1 cpp
# 或
.\scripts\switch-template.ps1 py
```

## 使用说明

### 导入题目流程

1. 在 VS Code 中打开 `AlgorithmSolves` 根目录（watcher 自动启动）
2. 在浏览器中打开题目页面（Codeforces、AtCoder 等）
3. 点击 Competitive Companion 扩展图标发送题目数据
4. VS Code 会弹出语言选择对话框
5. 选择 `C++` 或 `Python`
6. 题目代码文件会自动创建在 `src/cpp/` 或 `src/py/` 目录下
7. 模板会自动应用

### 注意事项

1. **watcher 自动启动**：配置了 `tasks.json`，打开工作区时自动启动，无需手动运行
2. **取消导入**：在语言选择对话框中点击取消或关闭，会显示 "Cancelled"，不会创建任何文件
3. **模板标识检查**：watcher 会检查文件内容是否已包含模板标识，避免重复覆盖

## 已知问题

| 问题 | 影响 | 解决方案 |
|------|------|----------|
| `cph-ng.problem.templateFile` 只能设置一个模板 | Python 文件创建时会先写入 C++ 模板，再被 watcher 覆盖 | 已通过 watcher 解决，无实际影响 |
| sandbox 限制 `fs` 模块 | 脚本无法直接写入文件 | 由 cph-ng 负责文件创建，脚本只返回路径 |
