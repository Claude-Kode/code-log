```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 100005;
int n;
int col[N];  // 每个节点的颜色
vector<int> g[N];

// 第一次 dfs，求子树大小和重儿子
int sz[N], son[N];
void dfs1(int u, int fa) {
  sz[u] = 1;
  int mx = 0;
  for (int v : g[u]) {
    if (v == fa) continue;
    dfs1(v, u);
    sz[u] += sz[v];
    if (sz[v] > mx) {
      mx = sz[v];
      son[u] = v;  // 重儿子
    }
  }
}

// 全局计数数组，以及当前不同颜色数
int cnt[N], nowAns;
int ans[N];  // 存每个节点的答案

// 暴力加/减 u 子树内所有节点的贡献
void add(int u, int fa, int val) {
  cnt[col[u]] += val;
  if (val == 1 && cnt[col[u]] == 1) nowAns++;   // 新颜色出现
  if (val == -1 && cnt[col[u]] == 0) nowAns--;  // 颜色消失
  for (int v : g[u]) {
    if (v == fa) continue;
    add(v, u, val);
  }
}

// 树上启发式合并主过程
// keep = true ：保留当前子树的计数数组
// keep = false：算完后清空
void dfs2(int u, int fa, bool keep) {
  // 1. 先处理所有轻儿子，并且不保留
  for (int v : g[u]) {
    if (v == fa || v == son[u]) continue;
    dfs2(v, u, false);
  }
  // 2. 处理重儿子，并且保留它的计数数组
  if (son[u]) {
    dfs2(son[u], u, true);
  }
  // 3. 暴力把轻儿子的子树加进来
  for (int v : g[u]) {
    if (v == fa || v == son[u]) continue;
    add(v, u, 1);
  }
  // 4. 加上 u 自己
  cnt[col[u]]++;
  if (cnt[col[u]] == 1) nowAns++;

  // 此时 cnt 数组里就是 u 整个子树的信息，记录答案
  ans[u] = nowAns;

  // 5. 如果当前是轻儿子，就整个清空
  if (!keep) {
    add(u, fa, -1);
  }
}

int main() {
  cin >> n;
  for (int i = 1; i <= n; i++) cin >> col[i];
  for (int i = 1; i < n; i++) {
    int u, v;
    cin >> u >> v;
    g[u].push_back(v);
    g[v].push_back(u);
  }

  dfs1(1, 0);        // 算重儿子
  dfs2(1, 0, true);  // 算答案（根节点保留不清空也可以）

  for (int i = 1; i <= n; i++) {
    cout << ans[i] << ' ';
  }
  return 0;
}
```
